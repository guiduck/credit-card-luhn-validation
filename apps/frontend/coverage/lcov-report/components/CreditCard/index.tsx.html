interface CreditCardProps {
  className: string;
  flipped: boolean;
  holder: string;
  number: string;
  expiration: string;
  cvc: string;
}

  className,
  flipped,
  holder,
  number,
  expiration,
  cvc,

  );

          <div className="absolute top-[70%] left-[5%]">

          {cvc}.
